130double Vector::phi() const {
131 double r = sqrt( x * x + y * y );
132 if( r == 0 ) {
133 if( z > 0 )
134 return PI / 2;
135 else if( z < 0 )
136 return -PI / 2;
137 else
138 return 0;
139 } else
140 return atan( z / r );
141}
